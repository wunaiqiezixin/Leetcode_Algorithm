```C
#define _CRT_SECURE_NO_WARNINGS
#include<stdio.h>
int max(int a, int b)
{
	return a > b ? a : b;
}
int nums[1000];
int maxLen[1000];

int main(void)
{
	int n;
	scanf("%d", &n);
	int ans = 1;
	int i, j;
	for (i = 0; i < n; i++)
	{
		scanf("%d", &nums[i]);
		maxLen[i] = 1;
	}
	for (i = 1; i < n; i++)
	{
		for (j = 0; j < i; j++)
		{
			if (nums[i] > nums[j])
			{
				maxLen[i] = max(maxLen[i], maxLen[j] + 1);
				if (maxLen[i] > ans) ans = maxLen[i];
			}
		}
	}
	printf("%d", ans);
}
```
