## 枚举法的思想
* a的范围 [2,N]
* b的范围 [2,a-1]
* c的范围 [b,a-1]
* d的范围 [c,a-1]
### ***C***
```C
#define _CRT_SECURE_NO_WARNINGS
#include<stdio.h>
#define cube(x) (x)*(x)*(x)

int main(void)
{
	int N;
	scanf("%d", &N);
	int a, b, c, d;
	/* b<=c<=d */
	for (a = 2; a <= N; a++)
	{
		for (b = 2; b < a; b++)
		{
			for (c = b; c < a; c++)
			{
				for (d = c; d < a; d++)
					if (cube(a) == cube(b) + cube(c) + cube(d))
						printf("Cube = %d, Triple = (%d,%d,%d)\n",a,b,c,d);
			}
		}
```
