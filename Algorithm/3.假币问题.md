#### 字符串函数--include<string.h>
* **原型** char* strchr(const char * s,int c);
>如果s字符串中包含c字符，该函数返回指向s字符串中首次出现的c字符的指针(末尾的空字符也是字符串的一部分，所以在查找范围内)；如果在字符串s中未找到c字符，该函数返回空指针


## 枚举法
- 判断A是否为重的假币
- 若不成立，再判断A是否为较轻的假币
- 依次枚举到L

### ***C***
```C
#define _CRT_SECURE_NO_WARNINGS
#include<stdio.h>
#define false 0
#define true 1
//用二维数组存放一组结果（三次称量）的字符串
char Left[3][7];
char Right[3][7];
char result[3][5];
//函数isFake用于判断字符c是不是轻的/重的假币
_Bool isFake(char c, _Bool isLight);

int main(void)
{
	int n;
	scanf("%d", &n);
	while (n--)//读取三次称量的结果
	{
		for (int i = 0; i < 3; i++)
		{
			scanf("%s %s %s", &Left[i], &Right[i], &result[i]);
		}
		char c; //枚举银币'A'~'L'，依次判断是否是假币
		for (c = 'A'; c <= 'L'; c++)
		{
			if (isFake(c, true))//是否是较轻的假币
			{
				printf("%c is the counterfeit coin and it is light.\n", c);
				break;
			}
			else if (isFake(c,false)) //是否是较重的假币
			{
				printf("%c is the counterfeit coin and it is heavy.\n", c);
				break;
			}
		}
	}
}

_Bool isFake(char c, _Bool isLight)
{
	//两个指针分别指向每一次称量时左右两边的字符串
	char* pLeft, * pRight;
	//通过三次称量，判断c是否为假币
	for (int i = 0; i < 3; i++)
	{
		if (isLight)//判断是否为较轻的假币
		{
			pLeft = Left[i];
			pRight = Right[i];
		}
		else//重币与轻币的判断情况恰好相反
		{
			pLeft = Right[i];
			pRight = Left[i];
		}
		switch (result[i][0])
		{
		case 'u'://天平右边高
			if (strchr(pRight, c) == NULL)
			{
				return false;
			}
			break;
		case 'd'://天平左边高
		{
			if (strchr(pLeft, c) == NULL)
			{
				return false;
			}
			break;
		}
		case 'e'://天平平衡
			if (strchr(pLeft, c) || strchr(pRight, c))
			{
				return false;
			}
			break;
		}
	}
	return true;
}
```
