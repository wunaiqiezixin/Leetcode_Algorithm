# 381.领养宠物
[题目链接](https://kamacoder.com/problempage.php?pid=1461)

>排序 + 多指针


### ***C++***
```CPP
#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;
struct Pet
{
    int order;//编号
    int c1;//毛色1
    int c2;//毛色2
    bool operator<(const Pet& other) const 
    {
        return order < other.order;
    }//按编号从小到大排序
};
int main()
{
    //优化输入输出
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    //读入数据
    int n;
    cin >> n;
    vector<Pet> pets(n);//待领养的宠物
    for (int i = 0; i < n; ++i) cin >> pets[i].order;
    for (int i = 0; i < n; ++i) cin >> pets[i].c1;
    for (int i = 0; i < n; ++i) cin >> pets[i].c2;
    //排序
    sort(pets.begin(), pets.end());
    vector<bool> adopted(n, false);//是否已经领养
    int ptrs[4]{0, 0, 0, 0};//指向每一种毛色
    vector<int> colors[4];//每一种毛色对应的宠物
    for (int i = 0; i < n; ++i)
    {
        colors[pets[i].c1].push_back(i);
        if (pets[i].c2 != pets[i].c1)
            colors[pets[i].c2].push_back(i);
    }
    int m;
    cin >> m;
    vector<int> pref(m);
    for (int i = 0; i < m; ++i) cin >> pref[i];
    for (int i = 0; i < m; ++i)//第i位领养
    {
        int c = pref[i];
        bool flag = false;//领养是否成功
        while (ptrs[c] < colors[c].size())
        {
            int idx = colors[c][ptrs[c]];
            ptrs[c]++;
            if (!adopted[idx])
            {
                adopted[idx] = true;
                flag = true;
                cout << pets[idx].order << (i == m - 1 ? "" : " ");
                break;
            }
        }
        if (!flag)
            cout << -1 << (i == m - 1 ? "" : " ");
    }
}
```
