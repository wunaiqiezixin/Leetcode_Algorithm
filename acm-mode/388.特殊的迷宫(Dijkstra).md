# 388.特殊的迷宫
[题目链接](https://kamacoder.com/problempage.php?pid=1468)

---

**Dijkstra + 剪枝优化**  


### ***C++***
```cpp
#include <iostream>
#include <vector>
#include <queue>
#include <cstdint>
using namespace std;
#define INF 2e18
#define MAX_VAL 200005
/*普通移动的方向数组*/
int dir[4][2]{-1,0,0,1,0,-1,1,0};

struct comparison {
    bool operator()(const pair<int, uint64_t>& a, 
                    const pair<int, uint64_t>& b) const {
                        return a.second > b.second;
                    }
};

void solve() {
    int n, m;
    cin >> n >> m;
    const int MAX_COST = n + m;//最大花费不会超过n+m 
    const int cell_nums = n * m;
    vector<int> a(cell_nums);//n*m个数
    /* 每一个数对应的下标，如果不存在这个数则为-1 */
    vector<uint64_t> val_to_idx(MAX_VAL, -1);

    for (int i = 0; i < cell_nums; ++i) {
        cin >> a[i];
        val_to_idx[a[i]] = i;
    }

    //最短路径算法：dijkstra
    vector<uint64_t> minDist(cell_nums, INF);/*到源点的最短距离*/
    priority_queue<pair<int, uint64_t>,
                   vector<pair<int, uint64_t>>,
                   comparison> pq; /*小顶堆：<顶点编号，到源点的距离>*/

    int start = 0, end = cell_nums - 1;
    minDist[start] = 0;

    //将起点入队
    pq.push({start, minDist[start]});
    while (!pq.empty()) {
        //取距离源点最近的顶点
        auto [u, d] = pq.top();pq.pop();

        if (d > minDist[u]) continue;
        //如果u是终点，返回
        if (u == end) {
            cout << d << '\n';
            return;
        }

        //普通移动：
        int r = u / m, c = u % m;/*顶点所在行和列*/
        for (int i = 0; i < 4; ++i) {
            int rr = r + dir[i][0];
            int cc = c + dir[i][1];
            if (rr < 0 || cc < 0 || rr >= n || cc >= m)
                continue;
            
            //更新顶点距离源点的最近距离
            int v = rr * m + cc;/*rr行cc列对应的顶点编号*/
            if (minDist[v] > d + 1) {
                minDist[v] = d + 1;
                pq.push({v, minDist[v]});
            }
        }

        //特殊传送：
        for (int k = 0; k <= MAX_COST; ++k) {
            uint64_t target_val = k ^ a[u];//^运算的性质
            if (target_val >= MAX_VAL || val_to_idx[target_val] == -1) {
                //传送花费为k的目标值过大或不存在
                continue;
            }
            //从顶点u花费k可以传送到目标顶点v(隐藏的一条边)
            int v = val_to_idx[target_val];
            //更新最短距离
            if (minDist[v] > k + d) {
                minDist[v] = k + d;
                pq.push({v, minDist[v]});
            }
        }
    }
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int t;
    cin >> t;
    while (t--) {
        solve();
    }
    return 0;
}
```
