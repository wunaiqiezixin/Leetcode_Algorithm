### 🚀动态规划
- 明确子问题
>不能是：前n个数中的最长上升子列  
>应该是：以第n个数结尾的最长上升子列
- 明确状态
>每个子问题只有一个参数n，即状态就可以认为是n
- 找边界状态
>有的边界不需要计算，如数塔三角形的最下面一行  
>本题当n == 0时，就是边界状态
- 状态转移方程
```C
int nums[MAX];//题目给的数组
int maxLen[MAX];//子问题的解，初始化全部为1
/* 状态转移方程 */
for (int i = 1;i < MAX;i++)
{
    for (int j = 0;j < MAX;j++)
    {
        if (nums[i] > nums[j])
        {
            maxLen[i] = max(maxLen[i], maxLen[j] + 1);
        }
    }
}
```
### 完整代码
### ***C:***
```C
#define _CRT_SECURE_NO_WARNINGS
#include<stdio.h>
int max(int a, int b)
{
	return a > b ? a : b;
}
int nums[1000];
int maxLen[1000];

int main(void)
{
	int n;
	scanf("%d", &n);
	int ans = 1;
	int i, j;
	for (i = 0; i < n; i++)
	{
		scanf("%d", &nums[i]);
		maxLen[i] = 1;
	}
	for (i = 1; i < n; i++)
	{
		for (j = 0; j < i; j++)
		{
			if (nums[i] > nums[j])
			{
				maxLen[i] = max(maxLen[i], maxLen[j] + 1);
				if (maxLen[i] > ans) ans = maxLen[i];
			}
		}
	}
	printf("%d", ans);
}
```

### 引入tails数组
>tails[i]:长度为i+1的递增子序列中元素的最大值   
### 时间复杂度： $O(NlogN)$

### ***C++***
```CPP
#include <iostream>
#include <algorithm>
#include <vector>
using namespace std;
int main()
{
    //优化
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int n;
    cin >> n;
    //数组tails[i]:长度为i+1的递增子序列的最大值
    vector<int> tails;
    while (n--)
    {
        int num;
        cin >> num;
        //在tails中查找第一个大于等于num的元素
        auto it = lower_bound(tails.begin(), tails.end(), num);
        if (it == tails.end())
            tails.push_back(num);//num为最大值，可以接在后面使序列加长
        else 
            *it = num;//在该长度的递增子序列用num替换(num可能大于)
    }
    cout << tails.size();//tails的长度即为最长递增子序列的长度
    return 0;
}
```
