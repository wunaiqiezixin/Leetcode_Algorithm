# 018:别叫，这个大整数已经很简化了!
[题目链接](http://cxsjsxmooc.openjudge.cn/2023t3spring/018/)

### 完整代码
```CPP
#include <iostream> 
#include <cstring> 
#include <cstdlib> 
#include <cstdio> 
using namespace std;
const int MAX = 110; 
class CHugeInt {
// 在此处补充你的代码
private:
    char hugeInt[210];
    void reverse(char* str)/* 反转字符串 */
    {
        int i, j;
        for (i = 0, j = strlen(str) - 1; i < j; ++i, --j)
        {
            char tmp = str[i];
            str[i] = str[j];
            str[j] = tmp;
        }
    }
public:
    /* 构造函数 */
    CHugeInt(int n)
    {
        memset(hugeInt, 0, sizeof(hugeInt));
        sprintf(hugeInt, "%d", n);/* 向字符串中写入整数 */
        reverse(hugeInt);/* 反转字符串 */
    }
    CHugeInt(const char* str)
    {
        memset(hugeInt, 0, sizeof(hugeInt));
        strcpy(hugeInt, str);/* 拷贝字符串 */
        reverse(hugeInt);/* 反转字符串 */
    }


    /* 类对象 + 类对象 */
    CHugeInt operator+ (const CHugeInt & cur)
    {
        /* 初始化一个大整数对象 */
        CHugeInt temp(0);

        int carry = 0;//进位

        int i, curSum;
        for (i = 0; i < 210; ++i)
        {
            char c1 = hugeInt[i];
            char c2 = cur.hugeInt[i];
            if (c1 == 0 && c2 == 0 && carry == 0)
                break;
            if (c1 == 0) /* 遇到空字符 '\0' ，化为字符'0' */
                c1 = '0';
            if (c2 == 0)
                c2 = '0';
            curSum = (c1 - '0') + (c2 - '0') + carry;
            carry = curSum / 10;
            temp.hugeInt[i] = curSum % 10 + '0';
        }
        return temp;
    }
    /* 类对象 + 常数 */
    CHugeInt operator+ (int n)
    {
        CHugeInt temp(n);/* 将常数n初始化为一个大整数对象 */
        return *this + temp;/* 调用已经重载的+ */
    }
    /* 常数 + 类对象 */
    friend CHugeInt operator+ (int n, CHugeInt& cur)
    {
        return cur + n;/* 运用已经重载的+ */
    }

    /* 重载输出运算符 */
    friend ostream & operator<< (ostream& os, const CHugeInt& H)
    {
        int i;
        for (i = strlen(H.hugeInt) - 1; i >= 0; --i)
            os << H.hugeInt[i];
        return os;
    }

    /* 重载前置的++运算符 */
    CHugeInt operator++ ()
    {
        *this = *this + 1;
        return *this;
    }
    /* 重载后置的++运算符 */
    CHugeInt operator++ (int)
    {
        CHugeInt temp = *this;
        *this = *this + 1;
        return temp;
    }

    /* 重载   类对象 += 常数n */
    CHugeInt operator+= (int n)
    {
        *this = *this + n;
        return *this;
    }

};
int  main() 
{ 
	char s[210];
	int n;

	while (cin >> s >> n) {
		CHugeInt a(s);
		CHugeInt b(n);

		cout << a + b << endl;
		cout << n + a << endl;
		cout << a + n << endl;
		b += n;
		cout  << ++ b << endl;
		cout << b++ << endl;
		cout << b << endl;
	}
	return 0;
}
```
