# 93. 复原 IP 地址
[力扣题目链接](https://leetcode.cn/problems/restore-ip-addresses/description/)

---
## 🧠思路
用一个`startIndex`变量表示切割线(要插入点的下标)    
用一个`isValid()`函数判断分隔的每一段是否合法    
```CPP
    bool isValid(const string& s, int start, int end)//判断每一段是否合法
    {
        if (start > end) return false;
        if (s[start] == '0' && end > start) return false;
        int num = 0;
        for (int i = start; i <= end; i++)
        {
            if (s[i] < '0' || s[i] > '9') return false;
            num = num* 10 +  (s[i] - '0');
            if (num > 255) return false;
        }
        return true;
    }
```
回溯：`pintNum`表示已经插入的点的个数(树的深度)
```CPP
void backtracking(string& s, int startIndex, int pointNum);
```
注意当`pointNum == 3`时，我们只是判断了前两段是合法的，所以我们此时还要判断第三段的合法性，只有第三段也合法，我们才能收集答案
```CPP
        //终止条件：
        if (pointNum == 3)
        {
            //判断当前这一段是否合法
            if (isValid(s, startIndex, s.size() - 1))
            {
                //收集结果
                ans.push_back(s);
            }
            return;//树的深度最大为3
        }
```
单层搜索逻辑
```CPP
        for (int i = startIndex; i < s.size(); i++)
        {
            //子串为s[startIndex, i]
            if (isValid(s, startIndex, i))
            {
                //前序位置：插入点
                s.insert(s.begin() + i + 1, '.');
                pointNum++;
                backtracking(s, i + 2, pointNum);
                //后序位置：回溯
                s.erase(s.begin() + 1 + i);
                pointNum--;
            }
            else break;//不合法
        }
```
### ***CPP***
```CPP
class Solution {
private:
    bool isValid(const string& s, int start, int end)//判断每一段是否合法
    {
        if (start > end) return false;
        if (s[start] == '0' && end > start) return false;
        int num = 0;
        for (int i = start; i <= end; i++)
        {
            if (s[i] < '0' || s[i] > '9') return false;
            num = num* 10 +  (s[i] - '0');
            if (num > 255) return false;
        }
        return true;
    }
    vector<string> ans;//记录结果
    void backtracking(string& s, int startIndex, int pointNum)
    {
        //终止条件：
        if (pointNum == 3)
        {
            //判断当前这一段是否合法
            if (isValid(s, startIndex, s.size() - 1))
            {
                //收集结果
                ans.push_back(s);
            }
            return;//树的深度最大为3
        }
        //递归回溯
        for (int i = startIndex; i < s.size(); i++)
        {
            //子串为s[startIndex, i]
            if (isValid(s, startIndex, i))
            {
                //前序位置：插入点
                s.insert(s.begin() + i + 1, '.');
                pointNum++;
                backtracking(s, i + 2, pointNum);
                //后序位置：回溯
                s.erase(s.begin() + 1 + i);
                pointNum--;
            }
            else break;//不合法
        }
    }
public:
    vector<string> restoreIpAddresses(string s) {
        backtracking(s, 0, 0);
        return ans;
    }
};
```
