# 17. 电话号码的字母组合
[力扣题目链接](https://leetcode.cn/problems/letter-combinations-of-a-phone-number/description/)   

---
## 🧠思路
先用一个二维数组将数字与字母对应   
```C
const char pairs[10][5] = {
    "",     //0
    "",     //1
    "abc",  //2
    "def",  //3
    "ghi",  //4
    "jkl",  //5
    "mno",  //6
    "pqrs", //7
    "tuv",  //8
    "wxyz"  //9
};
```

抽象为树结构    
用一个```result```二维数组记录结果，```resultTop```表示长度   
用一个```path```数组记录路径，```pathTop```表示长度，同时也代表了树的深度   
先来看根节点，深度```pathTop==0```,根节点的孩子为数字```digits[pathTop]```对应的字母组，用一个for循环遍历它的孩子，在前序位置将字母加入路径，在后序位置将字母移除   
### ***C:***
```C
/**
 * Note: The returned array must be malloced, assume caller calls free().
 */
const char pairs[10][5] = {
    "",     //0
    "",     //1
    "abc",  //2
    "def",  //3
    "ghi",  //4
    "jkl",  //5
    "mno",  //6
    "pqrs", //7
    "tuv",  //8
    "wxyz"  //9
};
void backtracking(char* digits, char** result, int* resultTop, char* path, int pathTop)
{
    //终止条件：叶子节点
    if (pathTop == strlen(digits))
    {
        char* tmp = (char*)malloc(sizeof(char) * (pathTop + 1));
        path[pathTop] = '\0';//先在末尾添加空字符，变成字符串，才能用strcpy()函数
        strcpy(tmp, path);
        result[(*resultTop)++] = tmp;
        return;
    }
    //递归遍历
    int digit = digits[pathTop] - '0';
    for (int i = 0; i < strlen(pairs[digit]); i++)
    {
        path[pathTop++] = pairs[digit][i];//pathTop：深度，path数组长度
        backtracking(digits, result, resultTop, path, pathTop);
        pathTop--;//回溯
    }
    return;
}
char** letterCombinations(char* digits, int* returnSize) {
    char** result = (char**)malloc(sizeof(char*) * 256);
    char* path = (char*)malloc(sizeof(char) * 5);
    *returnSize = 0;
    backtracking(digits, result, returnSize, path, 0);
    return result;
}
```
### ***C++***
```CPP
class Solution {
private:
    const string pairs[10] = {
        "",//0
        "",//1
        "abc",//2
        "def",//3
        "ghi",//4
        "jkl",//5
        "mno",//6
        "pqrs",//7
        "tuv",//8
        "wxyz",//9
    };
    vector<string> ans;
    string path;
    void backtracking(const string& digits, int depth)
    {
        if (path.length() == digits.length())
        {
            ans.push_back(path);
            return;
        }
        int idx = digits[depth] - '0';
        string letter = pairs[idx];
        for (int i = 0; i < letter.length(); i++)
        {
            path.push_back(letter[i]);
            backtracking(digits, depth + 1);
            path.pop_back();
        }
    }
public:
    vector<string> letterCombinations(string digits) {
        backtracking(digits, 0);
        return ans;
    }
};
```
