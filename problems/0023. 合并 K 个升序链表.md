# 23. 合并 K 个升序链表
[力扣题目链接](https://leetcode.cn/problems/merge-k-sorted-lists/description/?envType=problem-list-v2&envId=heap-priority-queue)

---

## 思路
>优先队列

维护一个最小堆，先将所有头节点入堆   
获取堆的`top`节点，将其与哨兵节点连接起来  
再将'top`节点的下一个节点入堆(如果不为`nullptr`)

### ***C++***
```CPP
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
class Solution {
private:
    struct mycomparison 
    {
        bool operator() (ListNode* const & node1, ListNode* const & node2) const
        {
            return node1->val > node2->val;//最小堆
        }
    };
public:
    ListNode* mergeKLists(vector<ListNode*>& lists) {
        if (lists.empty())
            return nullptr;
        priority_queue<ListNode*, vector<ListNode*>, mycomparison> pri_que;
        //所有头节点入堆
        for (ListNode* head : lists)
        {
            if (head)
                pri_que.push(head);
        }
        //哨兵节点
        ListNode* dummy = new ListNode(-1);
        ListNode* cur = dummy;

        while (!pri_que.empty())
        {
            ListNode* node = pri_que.top();
            pri_que.pop();
            cur->next = node;
            cur = cur->next;
            //下一节点入堆
            if (node->next)
                pri_que.push(node->next);
        }
        return dummy->next;
    }
};
```
