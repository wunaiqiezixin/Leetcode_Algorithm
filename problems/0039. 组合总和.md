# 39. 组合总和
[力扣题目链接](https://leetcode.cn/problems/combination-sum/description/)

---
## 🧠思路
本题的特殊在于数组```candidates```中的元素可以重复选择，因此```startIndex```变量的使用会稍微有一点变化    
用一个二维数组```result```记录符合条件的结果，```resultTop```表示数组长度       
用一个一维数组```path```来记录遍历的路径，```pathTop```表示长度      
### ***C:***
```C
/**
 * Return an array of arrays of size *returnSize.
 * The sizes of the arrays are returned as *returnColumnSizes array.
 * Note: Both returned array and *columnSizes array must be malloced, assume caller calls free().
 */
//二维数组result记录结果
//一维数组path记录路径
//startIndex表示起始位置
void backtracking(int** result, int* resultTop, int* path , int pathTop, int target, int startIndex, int* candidates, int candidatesSize, int* colSizes)
{
    //如果和大于target返回
    if (target < 0) 
        return;
    //如果和等于target,记录结果
    if (target == 0)
    {
        //拷贝元素
        int* tmp = (int*)malloc(sizeof(int) * pathTop);
        for (int i = 0; i < pathTop; i++)
            tmp[i] = path[i];
        //收集结果
        result[(*resultTop)] = tmp;
        colSizes[(*resultTop)++] = pathTop;
        return;
    }
    //递归回溯
    for (int i = startIndex; i < candidatesSize; i++)
    {
        //前序位置：加入路径
        path[pathTop++] = candidates[i];
        backtracking(result, resultTop, path, pathTop, target - candidates[i], i, candidates, candidatesSize, colSizes);
        //后续位置：回溯
        pathTop--;
    }
    return;
}
int** combinationSum(int* candidates, int candidatesSize, int target, int* returnSize, int** returnColumnSizes) {
    //分配空间
    int** result = (int**)malloc(sizeof(int*) * 1001);
    int* path = (int*)malloc(sizeof(int) * 1001);
    int* colSizes = (int*)malloc(sizeof(int) * 1001);
    *returnSize = 0;
    backtracking(result, returnSize, path, 0, target, 0, candidates, candidatesSize, colSizes);
    *returnColumnSizes = colSizes;
    return result;
}
```
### ***C++***
```CPP
class Solution {
private:
    vector<vector<int>> ans;
    vector<int> path;
    void backtracking(vector<int>& candidates, int target, int startIndex)
    {
        //终止：
        if (target < 0) return;//剪枝
        if (target == 0)
        {
            ans.push_back(path);
            return;
        }

        for (int i = startIndex; i < candidates.size(); i++)
        {
            path.push_back(candidates[i]);
            backtracking(candidates, target - candidates[i], i);
            path.pop_back();
        }
        return;
    }
public:
    vector<vector<int>> combinationSum(vector<int>& candidates, int target) {
        backtracking(candidates, target, 0);
        return ans;
    }
};
```
