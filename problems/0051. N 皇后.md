# 51. N 皇后
[力扣题目链接](https://leetcode.cn/problems/n-queens/description/)

---
- 用一个for循环横向遍历一行的所有列
- 用递归回溯遍历每一行
### ***C++***
```CPP
class Solution {
private:
    vector<vector<string>> ans;//结果集
    //row表示处理棋盘的第row行,n表示棋盘的大小
    void backtracking(vector<string>& chessbord, int row, int n)
    {
        if (row == n) //当深度为n时，棋盘摆放完毕
        {
            ans.push_back(chessbord);
            return;
        }

        for (int col = 0; col < n; col++)
        {
            if (isValid(chessbord, row, col, n))//如果当前位置可以摆放皇后
            {
                chessbord[row][col] = 'Q';
                backtracking(chessbord, row + 1, n);
                chessbord[row][col] = '.';
            }
        }
        return;
    }
    bool isValid(vector<string>& chessbord, int row, int col, int n)//判断(row, col)是否可以摆放
    {
        //检查列
        for (int i = 0; i < row; i++)
        {
            if (chessbord[i][col] == 'Q')
                return false;
        }
        //检查左上方
        for (int i = row - 1, j = col - 1; i >= 0 && j >= 0; i-- , j--)
        {
            if (chessbord[i][j] == 'Q')
                return false;
        }
        //检查右上方
        for (int i = row - 1, j = col + 1; i >= 0 && j < n; i--, j++)
        {
            if (chessbord[i][j] == 'Q')
                return false;
        }
        return true;
    }
public:
    vector<vector<string>> solveNQueens(int n) {
        vector<string> chessbord(n, string(n, '.'));//棋盘 nxn
        backtracking(chessbord, 0, n);
        return ans;
    }
};
```
