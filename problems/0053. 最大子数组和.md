# 53. 最大子数组和
[力扣题目链接](https://leetcode.cn/problems/maximum-subarray/description/)

---
>这道题过于经典，可以很好的加深对算法的时间复杂度的理解

## 算法1：O($N^3$)
### ***C:***
```C
//算法1：O(n^3)
int maxSubArray(int* nums, int numsSize) {
    int maxSum = -10001, thisSum = 0;        //声明，不计入时间
    for (int i = 0; i < numsSize; i++)       //时间单位：N个
    {                                           
        for (int j = i; j < numsSize; j++)  //时间单位：N - i个 ， 表示成O ， 记成N个时间单位
        {
            thisSum = 0;
            for (int k = i; k <= j; k++)    //时间单位：j - i + 1 个 , 算成N个
            {   
                thisSum += nums[k];         
            }
            if (thisSum > maxSum) maxSum = thisSum;
        }
    }
    return maxSum;
}
```

int maxSubArray(int* nums, int numsSize) {

    {
        thisSum = 0;

        {
            thisSum += nums[j];
            if (thisSum > maxSum) maxSum = thisSum;
        }
    }
    return maxSum;

## 算法3：O(NlogN)
### ***C:***
```C
//算法3：O(NlogN)
int three_max(int a, int b, int c)
{
    int mx = a;
    if (b > mx) mx = b;
    if (c > mx) mx = c;
    return mx;
}
int getMax(int* nums, int left, int right)
{
    if (left >= right)
        return nums[left];
    int mid = left + (right - left) / 2;
    int leftMax = getMax(nums, left, mid);
    int rightMax = getMax(nums, mid + 1, right);
    int leftCrossMax = nums[mid] - 1;
    int rightCrossMax = nums[mid + 1] - 1;

    //N个：
    int sum = 0;
    for (int i = mid; i >= left; i--)
    {
        sum += nums[i];
        if (sum > leftCrossMax) leftCrossMax = sum;
    }
    sum = 0;
    for (int i = mid + 1; i <= right; i++)
    {
        sum += nums[i];
        if (sum > rightCrossMax) rightCrossMax = sum;
    }
    //
    return three_max(leftMax, rightMax, leftCrossMax + rightCrossMax);
}
int maxSubArray(int* nums, int numsSize) {
    return getMax(nums, 0, numsSize - 1);
}
```
## 算法4：O(N)
### ***C:***
```C
//动态规划的视角:
//子问题：以下标i结尾的最大和f(i)
//状态转移方程：
//     如果f(i - 1) > 0 ---> f(i) = nums[i] + f(i - 1)
//     如果f(i - 1) < 0 ---> f(i) = nums[i]

//算法4：O(N)
int maxSubArray(int* nums, int numsSize) {
    int maxSum = nums[0], thisSum = nums[0];
    for (int i = 1; i < numsSize; i++)
    {
        //状态转移
        thisSum = thisSum > 0 ? (nums[i] + thisSum) : nums[i];
        if (maxSum < thisSum) maxSum = thisSum;
    }
    return maxSum;
}
```
