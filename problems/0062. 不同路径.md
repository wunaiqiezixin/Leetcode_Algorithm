# 62. 不同路径
[力扣题目链接](https://leetcode.cn/problems/unique-paths/description/)

---
>Dynamic Programming
## 思路
状态：到达(i, j)的所有路径和为f(i, j)  
边界状态：i = 0 或 j = 0 时，f(i, j) = 1  
状态转移方程: f(i, j) = f(i - 1, j) + f(i, j - 1)  

>>Memoization Search

### ***C++***
```CPP
/* Memoization Search */
class Solution {
private:
    int memo_search(int i, int j, vector<vector<int>>& memo)
    {
        if (memo[i][j])
            return memo[i][j];
        memo[i][j] = memo_search(i - 1, j, memo) + memo_search(i, j - 1, memo);
        return memo[i][j];
    }
    
public:
    int uniquePaths(int m, int n) {
        vector<vector<int>> memo(m, vector<int>(n, 0));
        for (int i = 0; i < m; i++)
            memo[i][0] = 1;
        for (int i = 0; i < n; i++)
            memo[0][i] = 1;
        memo_search(m - 1, n - 1, memo);
        return memo[m - 1][n - 1];
    }
};
```

>>Recurrence

### ***C++***
```CPP
class Solution {
public:
    int uniquePaths(int m, int n) {
        vector<vector<int>> dp(m, vector(n, 0));
        /* base case */
        for (int i = 0; i < m; i++)
            dp[i][0] = 1;
        for (int i = 0; i < n; i++)
            dp[0][i] = 1;
        /* Fill the DP table using the recurrence relation */
        for (int i = 1; i < m; i++)
        {
            for (int j = 1; j < n; j++)
            {
                dp[i][j] = dp[i - 1][j] + dp[i][j - 1];
            }
        }
        return dp[m - 1][n - 1];
    }
};
```

>DFS(超时)
### ***C++***
```CPP
class Solution {
private:
    int dfs(int i, int j, int m, int n)
    {
        if (i > m || j > n || i <= 0 || j <= 0)
            return 0;
        if (i == m && j == n)
            return 1;
        return dfs(i + 1, j, m, n) + dfs(i, j + 1, m, n);
    }
public:
    int uniquePaths(int m, int n) {
        return dfs(1, 1, m, n);
    }
};
```
>Memo_DFS

class Solution {
private:
    int memo_dfs(int i, int j, int m, int n, vector<vector<int>>& memo)
    {
        if (i > m || j > n || i <= 0 || j <= 0)
            return 0;
        if (i == m && j == n)
            return 1;
        if (memo[i][j])
            return memo[i][j];
        memo[i][j] = memo_dfs(i + 1, j, m, n, memo) + memo_dfs(i, j + 1, m, n, memo);
        return memo[i][j];
    }
public:
    int uniquePaths(int m, int n) {
        vector<vector<int>> memo(m + 1, vector<int>(n + 1, 0));
        return memo_dfs(1, 1, m, n, memo);
    }
};
