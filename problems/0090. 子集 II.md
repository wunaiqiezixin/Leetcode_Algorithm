# 90. 子集 II
[力扣题目链接](https://leetcode.cn/problems/subsets-ii/description/)

---
- 每一个节点都要收集答案
- `nums`数组的元素可以重复，每个元素只能用一次
- 不能出现重复的组合
- 为了更好的去重，应该先对数组排序

用一个`used`数组去重

### ***C++***
```CPP
class Solution {
private:
    vector<vector<int>> ans;
    vector<int> path;
    void backtracking(vector<int>& nums, int startIndex, vector<bool>& used)
    {
        //收集答案
        ans.push_back(path);
        //终止
        if (startIndex >= nums.size()) return;
        //递归回溯
        for (int i = startIndex; i < nums.size(); i++)
        {
            /*
            used[i - 1] == true 表明在同一树枝
            used[i - 1] == false 表明在同一树层
            我们要对树层的元素去重
            */
            if (i > 0 && nums[i] == nums[i - 1] && used[i - 1] == false)
                continue;
            
            path.push_back(nums[i]);
            used[i] = true;
            backtracking(nums, i + 1, used);
            path.pop_back();
            used[i] = false;
        }
    }
public:
    vector<vector<int>> subsetsWithDup(vector<int>& nums) {
        vector<bool> used(nums.size(), false);
        //去重需要排序数组
        sort(nums.begin(), nums.end());
        backtracking(nums, 0, used);
        return ans;
    }
};
```

class Solution {
private:
    vector<vector<int>> ans;
    vector<int> path;
    void backtracking(vector<int>& nums, int startIndex)
    {
        //收集结果：
        ans.push_back(path);
        //终止：
        if (startIndex >= nums.size()) return;
        
        //用hash数组去重：
        int hash[21] = {0};
        for (int i = startIndex; i < nums.size(); i++)
        {
            if (!hash[nums[i] + 10])//如果在同一层还没使用
            {
                path.push_back(nums[i]);
                hash[nums[i] + 10] = 1;
                backtracking(nums, i + 1);
                path.pop_back();
            }
            //else continue;
        }
    }
public:
    vector<vector<int>> subsetsWithDup(vector<int>& nums) {
        //排序数组
        sort(nums.begin(), nums.end());
        backtracking(nums, 0);
        return ans;
    }
};
