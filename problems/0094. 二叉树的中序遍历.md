# 94. 二叉树的中序遍历
[力扣题目链接](https://leetcode.cn/problems/binary-tree-inorder-traversal/description/)

## 📝递归遍历二叉树
先遍历左子树，再遍历根节点，最后遍历右子树  
递归解决
### ***C:***
```C
/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     struct TreeNode *left;
 *     struct TreeNode *right;
 * };
 */
/**
 * Note: The returned array must be malloced, assume caller calls free().
 */

//递归解决：
void infixOrder(struct TreeNode* root, int* ret, int* retSize)
{
    //递归的边界
    if (root == NULL)
        return;
    //先遍历左子树
    infixOrder(root->left, ret, retSize);
    //再遍历根节点
    ret[(*retSize)++] = root->val;
    //最后遍历右子树
    infixOrder(root->right, ret, retSize);
}
int* inorderTraversal(struct TreeNode* root, int* returnSize) {
    *returnSize = 0;
    int* ret = (int*)malloc(sizeof(int) * 100);
    infixOrder(root, ret, returnSize);
    return ret;
}
```
## 🌼中序遍历--迭代法
迭代法遍历二叉树就是两个操作
###### 访问节点
###### 处理节点
与前序遍历不同的是，前序遍历访问节点与处理节点同时进行，即我先访问了根节点，我就可以将根节点的值加入数组  
但是中序遍历不同，访问了根节点，但这并不是我们要优先处理的点，我们只能将它放入栈  
因此，对于中序遍历，我们用一个栈来记录访问过的节点，用指针遍历并处理节点
### ***C:***
```C
/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     struct TreeNode *left;
 *     struct TreeNode *right;
 * };
 */
/**
 * Note: The returned array must be malloced, assume caller calls free().
 */
int* inorderTraversal(struct TreeNode* root, int* returnSize) {
    //定义一个栈，用来记录访问过的节点
    struct TreeNode** stack = (struct TreeNode**)malloc(sizeof(struct TreeNode*) * 100);
    int stackTop = 0;
    struct TreeNode* cur;//用来遍历二叉树
    //定义一个数组，用来存节点的值
    int* ret = (int*)malloc(sizeof(int) * 100);
    *returnSize = 0;
    //访问节点
    cur = root;
    while (cur || stackTop)
    {
      
        if (cur)/*左*/
        {
            stack[stackTop++] = cur;
            cur = cur->left;
        }
        else
        {
            //弹出栈顶元素
            cur = stack[--stackTop];/*中*/
            //处理栈顶元素的值，放入数组
            ret[(*returnSize)++] = cur->val;
            //cur指向栈顶元素的右孩子
            cur = cur->right;/*右*/
        }
    }
    return ret;
}
```
## 🚀非递归遍历的统一形式代码
- 二叉树的遍历包括两个操作
1. 访问节点
2. 处理节点 

- 需要用到一个数据结构：栈


###### 无法将代码形式进行统一的难点：
- 前序遍历：访问节点和处理节点的操作同时进行，即每访问到一个节点就将其值加入数组
- 后序遍历：可以由前序遍历代码修改并反转数组得到，即中左右-中右左-左右中，这倒是很好统一
- 中序遍历：难点就是它，访问节点和处理节点并不同时进行


#### 📝思路一：空指针标记法
对于访问过但并未处理过的节点，在这类节点入栈后再入栈一个空指针NULL，用于表示这个节点需要在下一次处理
这样，我们就完美的处理了前中后序遍历的差别

中序遍历代码：
### ***C:***
```C
/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     struct TreeNode *left;
 *     struct TreeNode *right;
 * };
 */
/**
 * Note: The returned array must be malloced, assume caller calls free().
 */
int* inorderTraversal(struct TreeNode* root, int* returnSize) {
    //栈
    struct TreeNode** stack = (struct TreeNode**)malloc(sizeof(struct TreeNode*) * 1000);
    int stackTop = 0;
    //数组存值
    int* ret = (int*)malloc(sizeof(int) * 1000);
    *returnSize = 0;
    //根节点入栈
    if (root == NULL) return ret;
    stack[stackTop++] = root;
    //访问过未处理的节点用空指针标记
    while (stackTop)
    {
        //取栈顶元素的值，如果为NULL,则处理下一节点
        struct TreeNode* top = stack[stackTop - 1];
        if (top)//不为NULL
        {
            //弹出栈顶元素，根据“左中右”再入栈，标记NULL
            stackTop--;
            if (top->right) stack[stackTop++] = top->right;/*右孩子入栈*/
            stack[stackTop++] = top; /*中*/
            stack[stackTop++] = NULL; //标记刚刚访问过且未处理的中节点
            if (top->left) stack[stackTop++] = top->left;/*左*/
        }
        else //为NULL
        {
            //处理下一个节点
            stackTop--;//弹出NULL
            struct TreeNode* cur = stack[--stackTop];//弹出要处理的节点
            ret[(*returnSize)++] = cur->val;//将其值加入数组
        }
    }
    return ret;
}
```

class Solution:

>迭代
```Python3
# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
#迭代法:能往左就往左，不能往左就往右
class Solution:
    def inorderTraversal(self, root: Optional[TreeNode]) -> List[int]:
        ans = []#存答案
        if not root:
            return ans
        stack = []
        cur = root #遍历树
        #迭代
        while len(stack) or cur:
            if (cur):
                stack.append(cur)
                cur = cur.left#能往左就往左
            else:
                #处理节点
                cur = stack.pop()
                ans.append(cur.val)
                #不能往左就往右
                cur = cur.right
        return ans 
```

# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
"""
空指针标记法：标记访问过但未处理的节点
"""
#左中右
class Solution:
    def inorderTraversal(self, root: Optional[TreeNode]) -> List[int]:
        if not root:
            return []
        ans = []
        stack = [root]
        while len(stack):
            cur = stack.pop()
            if (cur):
                if cur.right:
                    stack.append(cur.right)
                stack.append(cur)
                stack.append(None)
                if cur.left:
                    stack.append(cur.left)
            else:
                cur = stack.pop()
                ans.append(cur.val)
        return ans
