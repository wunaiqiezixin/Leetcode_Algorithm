# 96. 不同的二叉搜索树
[力扣题目链接](https://leetcode.cn/problems/unique-binary-search-trees/description/)

---
>dynamic programming

## 思路
状态：dp[i] 表示有i个节点的二叉搜索树的种类  
状态转移方程：  
dp[i] = 以1为根节点的种类数 + 以2为根节点的种类数 + ...... + 以k为根节点的种类数...+...  
以k为根节点的种类数 = dp[k - 1] * dp[i - k]  


>记忆化搜索
### ***C++***
```CPP
class Solution {
private:
    int memo_search(int i, vector<int>& memo)
    {
        if (i == 0 || i == 1)
            memo[i] = 1;
        if (memo[i])
            return memo[i];
        /* recurrence */
        for (int j = 1; j <= i; j++)
            memo[i] += memo_search(j - 1, memo) * memo_search(i - j, memo);
        return memo[i];
    }

public:
    int numTrees(int n) {
        vector<int> memo(n + 1); /* memo[i]：表示i个节点的不同二叉搜索树的种类 */
        memo_search(n, memo);
        return memo[n];
    }
};
```
>递推
```CPP
class Solution {
public:
    int numTrees(int n) {
        vector<int> dp(n + 1);
        dp[0] = dp[1] = 1;/* base case */

        for (int i = 2; i <= n; i++)
        {
            for (int j = 1; j <= i; j++)
                dp[i] += dp[j - 1] * dp[i - j]; 
        }
        return dp[n];
    }
};
```
