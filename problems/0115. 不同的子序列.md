# 115. 不同的子序列
[力扣题目链接](https://leetcode.cn/problems/distinct-subsequences/description/)

---

### ***C++***
```CPP
class Solution {
public:
    int numDistinct(string s, string t) {
        //dp[i][j]:  s[0, i - 1]中t[0, j - 1]的最大个数
        vector<vector<uint64_t> > dp(s.size() + 1, vector<uint64_t>(t.size() + 1, 0));
        //初始化
        int i, j;
        for (j = 1; j <= t.size(); ++j)
            dp[0][j] = 0;//s为空串
        for (i = 0; i <= s.size(); ++i)
            dp[i][0] = 1;//t为空串
        //状态转移
        for (i = 1; i <= s.size(); ++i)
            for (j = 1; j <= t.size(); ++j)
                if (s[i - 1] == t[j - 1])
                    dp[i][j] = dp[i - 1][j - 1] + dp[i - 1][j];
                else
                    dp[i][j] = dp[i - 1][j];
        return dp[s.size()][t.size()];
    }
};
```
