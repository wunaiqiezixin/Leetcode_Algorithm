# 122. 买卖股票的最佳时机 II
[力扣题目链接](https://leetcode.cn/problems/best-time-to-buy-and-sell-stock-ii/description/)

---
## 思路
第i天的状态只能由第(i - 1)天的状态转移而来  
第i天持有股票的最大利润 = max( 第i - 1天持有股票的最大利润， 第i-1天不持有股票的最大利润 + 第i天股票的价格)  
第i天不持有股票的最大利润 = max(第i-1天不持有股票的最大利润，第i-1天持有股票的最大利润 - 第i天股票的价格)  


>动态规划
### ***C++***
```CPP
class Solution {
public:
    int maxProfit(vector<int>& prices) {
        int len = prices.size();
        vector<vector<int> > dp(len, vector<int>(2, 0));
        /*
        dp[i][0]:第i天持有股票的最大利润
        dp[i][1]:第i天不持有股票的最大利润
        */
        dp[0][0] = 0 - prices[0];
        int i;
        for (i = 1; i < len; ++i)
        {
            dp[i][0] = max(
                dp[i - 1][0],//第(i - 1)天持有股票的最大利润
                dp[i - 1][1] - prices[i]//第(i - 1)天没有持有股票
            );
            dp[i][1] = max(
                dp[i - 1][1],//第(i - 1)天不持有股票
                dp[i - 1][0] + prices[i]//第(i - 1)天持有股票
            );
        }
        return dp[len - 1][1];
    }
};
```

>空间优化
```CPP
class Solution {
public:
    int maxProfit(vector<int>& prices) {
        int len = prices.size();
        vector<vector<int> > dp(2, vector<int>(2, 0));
        /*
        dp[i][0]:第i天持有股票的最大利润
        dp[i][1]:第i天不持有股票的最大利润
        */
        dp[0][0] = 0 - prices[0];
        int i;
        for (i = 1; i < len; ++i)
        {
            dp[i % 2][0] = max(
                dp[(i - 1) % 2][0],//第(i - 1)天持有股票的最大利润
                dp[(i - 1) % 2][1] - prices[i]//第(i - 1)天没有持有股票
            );
            dp[i % 2][1] = max(
                dp[(i - 1) % 2][1],//第(i - 1)天不持有股票
                dp[(i - 1) % 2][0] + prices[i]//第(i - 1)天持有股票
            );
        }
        return dp[(len - 1) % 2][1];
    }
};
```
