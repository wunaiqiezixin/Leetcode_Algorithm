# 123. 买卖股票的最佳时机 III
[力扣题目链接](https://leetcode.cn/problems/best-time-to-buy-and-sell-stock-iii/description/)  

---
> 动态规划

### ***C++***
```CPP
class Solution {
public:
    int maxProfit(vector<int>& prices) {
        /* 
        dp[i][j]:
          第i天
            j = 0 -> 什么都没做
            j = 1 -> 第一次持有股票
            j = 2 -> 第一次不持有股票
            j = 3 -> 第二次持有股票
            j = 4 -> 第二次不持有股票
          最大利润
        */
        int len = prices.size();
        vector<vector<int> > dp(len, vector<int>(5, 0));
        dp[0][1] = 0 - prices[0];
        dp[0][3] = 0 - prices[0];
        int i;
        for (i = 1; i < len; ++i)
        {
            dp[i][1] = max(dp[i - 1][1], dp[i - 1][0] - prices[i]);
            dp[i][2] = max(dp[i - 1][2], dp[i - 1][1] + prices[i]);
            dp[i][3] = max(dp[i - 1][3], dp[i - 1][2] - prices[i]);
            dp[i][4] = max(dp[i - 1][4], dp[i - 1][3] + prices[i]);
        }
        return max(dp[len - 1][2], dp[len - 1][4]);
    }
};
```

>空间优化
```CPP
class Solution {
public:
    int maxProfit(vector<int>& prices) {
        int len = prices.size();
        vector<int> dp(5, 0);
        dp[1] = 0 - prices[0];
        dp[3] = 0 - prices[0];
        int i;
        for (i = 1; i < len; ++i)
        {
            dp[1] = max(dp[1], dp[0] - prices[i]);
            dp[2] = max(dp[2], dp[1] + prices[i]);
            dp[3] = max(dp[3], dp[2] - prices[i]);
            dp[4] = max(dp[4], dp[3] + prices[i]);
        }
        return max(dp[2], dp[4]);
    }
};
```
