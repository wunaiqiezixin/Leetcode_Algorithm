# 139. 单词拆分
[力扣题目链接](https://leetcode.cn/problems/word-break/description/)

---
> 回溯算法(超时)
```CPP
class Solution {
private:
    bool backtracking(string& s, unordered_set<string>& wordSet, int startIndex)
    {
        if (startIndex >= s.size())/* 切到字符串尾部，成功 */
            return true;
        for (int i = startIndex; i < s.size(); ++i)
        {
            string word = s.substr(startIndex, i - startIndex + 1);//取切割得到的单词
            if (wordSet.find(word) != wordSet.end() && backtracking(s, wordSet, i + 1)) //成功就一直往下递归，遇到不成功就返回一层，再往下
                return true;
        }
        /* 最顶层 */
        return false;//第二层以下均不成功，返回false
    }
public:
    bool wordBreak(string s, vector<string>& wordDict) {
        /* 哈希表 */
        unordered_set<string> wordSet(wordDict.begin(), wordDict.end());
        /* 回溯搜索，判断在s的所有切法中是否存在一种使得均在集合wordSet中 */
        return backtracking(s, wordSet, 0);
    }
};
```
>回溯算法(优化)

```CPP
class Solution {
private:
    bool backtracking(string& s, unordered_set<string>& wordSet, vector<bool>& memory, int startIndex)
    {
        if (startIndex >= s.size())
            return true;
        if (!memory[startIndex])
            return memory[startIndex];
        for (int i = startIndex; i < s.size(); ++i)
        {
            string str = s.substr(startIndex, i - startIndex + 1);
            if (wordSet.find(str) != wordSet.end() && backtracking(s, wordSet, memory, i + 1))
                return true;
        }
        memory[startIndex] = false;
        return false;
    }
public:
    bool wordBreak(string s, vector<string>& wordDict) {
        unordered_set<string> wordSet(wordDict.begin(), wordDict.end());
        vector<bool> memory(s.size(), 1);
        return backtracking(s, wordSet, memory, 0);
    }
};
```
>动态规划
```CPP
class Solution {
public:
    bool wordBreak(string s, vector<string>& wordDict) {
        //哈希表
        unordered_set<string> wordSet(wordDict.begin(), wordDict.end());
        //dp[i]: 前i个字符，能不能用字典中的单词拼出来
        vector<bool> dp(s.size() + 1, false);
        //前0个字符(空字符)
        dp[0] = true;
        //递推：如果dp[j] == true, 并且s的子串[j, i]在字典里面，那么dp[i] = true
        int i, j;
        for (i = 1; i <= s.size(); ++i)
        {
            for (j = 0; j < i; ++j)
            {
                string word = s.substr(j, i - j); //注意长度不是: i - j + 1
                if (dp[j] && wordSet.find(word) != wordSet.end())
                    dp[i] = true;
            }
        }
        return dp[s.size()];
    }
};
```
