# 144. 二叉树的前序遍历
[力扣题目链接](https://leetcode.cn/problems/binary-tree-preorder-traversal/description/)
---
## 📝递归前序遍历二叉树
先遍历根节点，再遍历左子树，最后遍历右子树  
非常适合用递归解决  
### ***C:***
```C
/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     struct TreeNode *left;
 *     struct TreeNode *right;
 * };
 */
/**
 * Note: The returned array must be malloced, assume caller calls free().
 */
//重新定义一个函数，以便递归
void preOrder(struct TreeNode* root, int* ret, int *retSize)
{
    if (root == NULL)//递归的边界
        return;
    ret[(*retSize)++] = root->val;
    preOrder(root->left, ret, retSize);
    preOrder(root->right, ret, retSize);
}

int* preorderTraversal(struct TreeNode* root, int* returnSize) {
    *returnSize = 0;
    int* ret = (int*)malloc(sizeof(int) * 100);
    preOrder(root, ret, returnSize);
    return ret;
}
```
## 📑非递归前序遍历二叉树
C语言实现递归依靠的是栈  
所以我们可以依靠栈来完成对二叉树的非递归前序遍历  
注意：  
根据栈先入后出的特点  
我们应该先判断节点有没有右孩子，再判断节点有没有左孩子  
### ***C:***
```C
/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     struct TreeNode *left;
 *     struct TreeNode *right;
 * };
 */
/**
 * Note: The returned array must be malloced, assume caller calls free().
 */
//迭代法中序遍历二叉树：中左右
/*
1. 根节点入栈
2. 弹出栈顶节点
3. 先判断节点有无‘右孩子’，如果有-->入栈
4. 再判断节点有无‘左孩子’，如果有-->入栈
5. 弹出栈顶节点，重复3--5，直到栈为空
*/
int* preorderTraversal(struct TreeNode* root, int* returnSize) {
    //定义一个数组用来存遍历到的元素
    *returnSize = 0;
    int* ret = (int*)malloc(sizeof(int) * 100);
    if (root == NULL) return ret;
    //定义一个栈
    struct TreeNode** stack = (struct TreeNode**)malloc(sizeof(struct TreeNode*) * 100);
    int stackTop = 0;
    //将根节点入栈
    stack[stackTop++] = root;
    //循环迭代
    while (stackTop)
    {
        //弹出栈顶元素,并将其值加入到数组中
        struct TreeNode* cur = stack[--stackTop];
        ret[(*returnSize)++] = cur->val;
        //先判断是否有右孩子
        if (cur->right)
            stack[stackTop++] = cur->right;
        //判断是否有左孩子
        if (cur->left)
            stack[stackTop++] = cur->left;
    }
    return ret;
}
```

## 🚀空指针标记法(前中后序统一)
```C
/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     struct TreeNode *left;
 *     struct TreeNode *right;
 * };
 */
/**
 * Note: The returned array must be malloced, assume caller calls free().
 */
//空指针标记法:中左右
int* preorderTraversal(struct TreeNode* root, int* returnSize) {
    *returnSize = 0;
    int* result = (int*)malloc(sizeof(int) * 101);
    if (root == NULL) return result;
    struct TreeNode** stack = (struct TreeNode**)malloc(sizeof(struct TreeNode*) * 101);
    int stackTop = 0;
    stack[stackTop++] = root;
    while (stackTop)
    {
        struct TreeNode* cur = stack[stackTop - 1];
        if (cur != NULL)
        {
            stackTop--;
            if (cur->right) stack[stackTop++] = cur->right;
            if (cur->left) stack[stackTop++] = cur->left;
            stack[stackTop++] = cur;
            stack[stackTop++] = NULL;
        }
        else
        {
            stackTop--;
            cur = stack[--stackTop];
            result[(*returnSize)++] = cur->val;
        }
    }
    return result;
}
```

```Python3
class Solution:
    def preorderTraversal(self, root: Optional[TreeNode]) -> List[int]:
        ans = []
        if not root:
            return ans
        stack = [root]
        while len(stack):
            node = stack[-1]
            stack.pop()
            ##先处理中节点
            ans.append(node.val)
            if node.right:
                stack.append(node.right)
            if node.left:
                stack.append(node.left)
        return ans
```
>统一代码形式
```Python3
#空指针标记法：
"""
怎么把前中后序遍历的代码统一呢？
也就是说，我掌握了一种就会写三种
遍历二叉树就做两件事：
1. 访问节点
2. 处理节点
三种遍历方式不同就是因为访问和处理节点并不一定同时进行
做法：
当访问到一个节点时，但是我们不是现在要处理的节点，我们就用None标记
"""
class Solution:
    def preorderTraversal(self, root: Optional[TreeNode]) -> List[int]:
        ans = []#记录结果
        if not root:
            return ans
        stack = [root]  #栈
        #迭代
        while len(stack):
            #取栈顶节点
            cur = stack.pop()
            if cur:#如果不为空
                if cur.right:   #右
                    stack.append(cur.right)
                if cur.left:    #左
                    stack.append(cur.left)
                stack.append(cur) #中
                stack.append(None) #中为下一次要处理的节点，用None标记
            else:#如果为空
                node = stack.pop()
                ans.append(node.val)
        return ans
```
