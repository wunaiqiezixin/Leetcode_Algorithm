# 145. 二叉树的后序遍历
[力扣题目链接](https://leetcode.cn/problems/binary-tree-postorder-traversal/description/)
---
## 📑递归后序遍历二叉树
先遍历左子树，再遍历右子树，最后遍历根节点  
递归解决
### ***C:***
```C
/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     struct TreeNode *left;
 *     struct TreeNode *right;
 * };
 */
/**
 * Note: The returned array must be malloced, assume caller calls free().
 */
//先遍历左子树，再遍历右子树，最后遍历根节点
//递归遍历：
void postOrder(struct TreeNode* root, int* ret, int* retSize)
{
    //递归的边界
    if (root == NULL)
        return;
    //遍历左子树
    postOrder(root->left, ret, retSize);
    //遍历右子树
    postOrder(root->right, ret, retSize);
    //遍历根节点
    ret[(*retSize)++] = root->val;
}
int* postorderTraversal(struct TreeNode* root, int* returnSize) {
    *returnSize = 0;
    int* ret = (int*)malloc(sizeof(int) * 100);
    postOrder(root, ret, returnSize);
    return ret;
}
```
## 📝迭代法后序遍历二叉树
迭代法后序(左右中)遍历二叉树，基于前序遍历(中左右)  
我们只需交换[前序迭代法遍历二叉树](https://github.com/wunaiqiezixin/Leetcode_Algorithm/blob/main/problems/0144.%20%E4%BA%8C%E5%8F%89%E6%A0%91%E7%9A%84%E5%89%8D%E5%BA%8F%E9%81%8D%E5%8E%86.md)的两行代码，就可以得到(中右左)  
在(中右左)的基础上，我们在反转数组元素，就得到了(左右中)
```C
/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     struct TreeNode *left;
 *     struct TreeNode *right;
 * };
 */
/**
 * Note: The returned array must be malloced, assume caller calls free().
 */
//迭代法：
//后序遍历是：左-右-中
//前序遍历是：中-左-右
/*
思路：
1. 在前序遍历代码的基础上调整一下代码顺序
2. 即：
    前序遍历(中-左-右)--->调整代码(中-右-左)--->反转数组(左-右-中)
*/
//reverse()用于反转数组：(中-左-右)--->(左-右-中)
void reverse(int* ar, int left, int right)
{
    while (left < right)
    {
        int tmp = ar[left];
        ar[left++] = ar[right];
        ar[right--] = tmp;
    }
}
int* postorderTraversal(struct TreeNode* root, int* returnSize) {
    //定义一个数组(栈)，来存树的节点
    struct TreeNode** stack = (struct TreeNode**)malloc(sizeof(struct TreeNode*) * 100);
    int stackTop = 0;//栈顶
    //定义一个数组，用于存遍历到的值
    int* ret = (int*)malloc(sizeof(int) * 100);
    *returnSize = 0;
    /***下面前序遍历树(调整代码顺序)***/
    //如果为空树，返回空数组
    if (root == NULL) return ret;
    //根节点入栈
    stack[stackTop++] = root;
    //循环迭代，直到栈为空
    while (stackTop)
    {
        //将根节点出栈，并将其值加入ret
        struct TreeNode* cur = stack[--stackTop];
        ret[(*returnSize)++] = cur->val;
        //判断根节点有无左孩子(先)
        if (cur->left)
            stack[stackTop++] = cur->left;
        //判断根节点有无右孩子(后)
        if (cur->right)
            stack[stackTop++] = cur->right;
        //对栈顶元素重复上面的操作
    }
    //反转数组
    reverse(ret, 0, (*returnSize) - 1);
    return ret;
}
```
## 🚀前中后序统一代码(空指针标记法)
```C
/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     struct TreeNode *left;
 *     struct TreeNode *right;
 * };
 */
/**
 * Note: The returned array must be malloced, assume caller calls free().
 */
int* postorderTraversal(struct TreeNode* root, int* returnSize) {
    int* ans = (int*)malloc(sizeof(int) * 1001);
    *returnSize = 0;
    if (root == NULL) return ans;
    struct TreeNode** stack = (struct TreeNode**)malloc(sizeof(struct TreeNode*) * 1001);
    int stackTop = 0;
    stack[stackTop++] = root;
    while (stackTop)
    {
        struct TreeNode* cur = stack[stackTop - 1];
        if (cur)
        {
            stackTop--;
            stack[stackTop++] = cur;
            stack[stackTop++] = NULL;
            if (cur->right) stack[stackTop++] = cur->right;
            if (cur->left) stack[stackTop++] = cur->left;
        }
        else
        {
            stackTop--;
            cur = stack[--stackTop];
            ans[(*returnSize)++] = cur->val;
        }
    }
    return ans;
}
```
