# 46. 携带研究材料
[题目链接](https://kamacoder.com/problempage.php?pid=1046)

---
## 思路 
经典的`0-1背包`问题  
`dp[i][j]`表示在空间不超过`j`的前提下,从`[0,i]`中取物品,能得到的最大价值  
状态转移：  
`dp[i][j]` = max( 取第i件物品得到的价值 , 不取第i件物品得到的价值 )  
`dp[i][j] = max( dp[i - 1][j] + dp[i - 1][j - space[i]] + value[i]`  
这里要注意，第i件物品所占的空间可能超过`j`，所以有可能不能取第`i`件物品  

### ***C++***
>二维dp
```CPP
#include<iostream>
#include<vector>
using namespace std;

int main(void)
{
    int M, N, i, j;
    cin >> M >> N;

    vector<int> space(M);//每件物品所占空间
    vector<int> value(M);//每件物品的价值
    for (i = 0; i < M; i++)
        cin >> space[i];
    for (i = 0; i < M; i++)
        cin >> value[i];
    
    vector<vector<int>> dp(M, vector<int>(N + 1, 0));
    /* 初始化dp数组 */
    for (i = space[0]; i <= N; i++)
        dp[0][i] = value[0];
    
    /* 遍历dp数组 */
    for (i = 1; i < M; i++)
    {
        for (j = 1; j <= N; j++)
        {
            if (j >= space[i])
                dp[i][j] = max(dp[i - 1][j], dp[i - 1][j - space[i]] + value[i]);
            else
                dp[i][j] = dp[i - 1][j];
        }
    }

    cout << dp[M - 1][N];

    return 0;
}
```

>一维dp
```CPP
#include<iostream>
#include<vector>
using namespace std;

int main(void)
{
    int M, N, i, j;
    cin >> M >> N;

    vector<int> space(M);//每件物品所占空间
    vector<int> value(M);//每件物品的价值
    for (i = 0; i < M; i++)
        cin >> space[i];
    for (i = 0; i < M; i++)
        cin >> value[i];
    
    vector<int> dp(N + 1, 0);

    for (i = 0; i < M; i++)
    {
        for (j = N; j >= space[i]; j--)
        {
            dp[j] = max(dp[j], dp[j - space[i]] + value[i]);
        }
    }

    cout << dp[N];

    return 0;
}
```
