# 215. 数组中的第K个最大元素
[力扣题目链接](https://leetcode.cn/problems/kth-largest-element-in-an-array/description/)

---
>Top-K问题
## 思路
用一个优先队列(小顶堆)    
从头到尾遍历数组，数组元素入堆，当堆的长度大于k时，弹出堆顶元素  
最后堆中剩余的即为最大的k个元素，堆顶即为第k大的元素   

### ***C++***
```CPP
class Solution {
public:
    int findKthLargest(vector<int>& nums, int k) {
        priority_queue<int, vector<int>, greater<int>> pri_que;//小顶堆
        //遍历数组，入堆，当堆的长度大于K时，弹出堆顶元素
        for (int i = 0; i < nums.size(); i++)
        {
            pri_que.push(nums[i]);
            if (pri_que.size() > k) 
                pri_que.pop();
        }
        return pri_que.top();
    }
};
```

想办法将前k大的元素放到数组的最左边，第k大的元素为`nums[k - 1]`

```CPP
class Solution {
private:
    void swap(int& a, int& b)
    {
        int tmp = a;
        a = b;
        b = tmp;
    }
    void quickSelect(vector<int>& nums, int start, int end, int k)
    {
        if (start >= end) 
            return;
        int left = start, right = end;
        int base = nums[left];
        while (left != right)
        {
            while (left < right && nums[right] <= base)
                right--;
            swap(nums[left], nums[right]);
            while (left < right && nums[left] >= base)
                left++;
            swap(nums[left], nums[right]);
        }
        int size = left - start + 1;
        if (size > k)
            quickSelect(nums, start, left - 1, k);
        else if (size < k)
            quickSelect(nums, left + 1, end, k - size);
        else 
            return;
    }
public:
    int findKthLargest(vector<int>& nums, int k) {
        quickSelect(nums, 0, nums.size() - 1, k);
        return nums[k - 1];
    }
};
```

### ***C***
```C

struct HeapStruct;
typedef struct HeapStruct* PriorityQueue;
struct HeapStruct
{
	int Capacity;
	int Size;
	int* Elements;
};

PriorityQueue
Initialize(int MaxElements)
{
	PriorityQueue H;

	H = malloc(sizeof(struct HeapStruct));

	/*Allocate the array plus one extra for sentinel*/
	H->Elements = malloc((MaxElements + 1) * sizeof(int));

	H->Capacity = MaxElements;
	H->Size = 0;
	H->Elements[0] = INT_MIN;
	
	return H;
}


int 
FindMin(PriorityQueue H)
{
	return H->Elements[1];
}

int
IsEmpty(PriorityQueue H)
{
	if (H == NULL || H->Size == 0 || H->Elements == NULL)
		return 1;
	return 0;
}


void 
Insert(int X, PriorityQueue H)
{
	/* H->Elements[0] is a sentinel */
	int i;

	for (i = ++H->Size; X < H->Elements[i / 2]; 
		i /= 2)
	{
		H->Elements[i] = H->Elements[i / 2];
	}
	H->Elements[i] = X;
}

int
DeleteMin(PriorityQueue H)
{
	int i, child;
	int MinElement, LastElement;

	if (IsEmpty(H))
	{
		return INT_MAX;
	}
	MinElement = FindMin(H);
	LastElement = H->Elements[H->Size--];

	for (i = 1; i * 2 <= H->Size; i = child)
	{
		/* Find smaller child */
		child = 2 * i;
		if (child != H->Size && H->Elements[child + 1] < 
			H->Elements[child])
			child++;

		/* percolate one level */
		if (LastElement > H->Elements[child])
			H->Elements[i] = H->Elements[child];
		else
			break;
	}
	H->Elements[i] = LastElement;
	return MinElement;
}


int findKthLargest(int* nums, int numsSize, int k) {
    PriorityQueue H = Initialize(k + 1);
    for (int i = 0; i < numsSize; i++)
    {
        Insert(nums[i], H);
        if (H->Size > k)
            DeleteMin(H);
    }
    return FindMin(H);
}
```
