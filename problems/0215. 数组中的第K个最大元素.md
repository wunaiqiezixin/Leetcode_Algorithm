# 215. 数组中的第K个最大元素
[力扣题目链接](https://leetcode.cn/problems/kth-largest-element-in-an-array/description/)

---
>Top-K问题
## 思路
用一个优先队列(小顶堆)    
从头到尾遍历数组，数组元素入堆，当堆的长度大于k时，弹出堆顶元素  
最后堆中剩余的即为最大的k个元素，堆顶即为第k大的元素   

### ***C++***
```CPP
class Solution {
public:
    int findKthLargest(vector<int>& nums, int k) {
        priority_queue<int, vector<int>, greater<int>> pri_que;//小顶堆
        //遍历数组，入堆，当堆的长度大于K时，弹出堆顶元素
        for (int i = 0; i < nums.size(); i++)
        {
            pri_que.push(nums[i]);
            if (pri_que.size() > k) 
                pri_que.pop();
        }
        return pri_que.top();
    }
};
```

```CPP
class Solution {
private:
    void swap(int& a, int& b)
    {
        int tmp = a;
        a = b;
        b = tmp;
    }
    void quickSelect(vector<int>& nums, int start, int end, int k)
    {
        if (start >= end) 
            return;
        int left = start, right = end;
        int base = nums[left];
        while (left != right)
        {
            while (left < right && nums[right] <= base)
                right--;
            swap(nums[left], nums[right]);
            while (left < right && nums[left] >= base)
                left++;
            swap(nums[left], nums[right]);
        }
        int size = left - start + 1;
        if (size > k)
            quickSelect(nums, start, left - 1, k);
        else if (size < k)
            quickSelect(nums, left + 1, end, k - size);
        else 
            return;
    }
public:
    int findKthLargest(vector<int>& nums, int k) {
        quickSelect(nums, 0, nums.size() - 1, k);
        return nums[k - 1];
    }
};
```

```C
struct HeapStruct;
typedef struct HeapStruct* PriorityQueue;

struct HeapStruct {
    int Capacity;
    int Size;
    int* Elements;
};

PriorityQueue Initialized(int maxElements)
{
    PriorityQueue H = malloc(sizeof(struct HeapStruct));
    //todo: H == NULL

    H->Capacity = maxElements;
    H->Size = 0;
    H->Elements = malloc(sizeof(int) * (maxElements + 1));
    //todo: H->Elements == NULL
    H->Elements[0] = INT_MIN;
    return H;
}

void Insert(int x, PriorityQueue H)
{
    //todo: priority queue is full 

    /* Percolate up */
    int i;
    for (i = ++H->Size; x < H->Elements[i / 2]; i /= 2)
        H->Elements[i] = H->Elements[i / 2];
    H->Elements[i] = x;
}

int DeleteMin(PriorityQueue H)
{
    //todo: priority queue is empty

    /* Percolate down */
    int i, child;
    int MinElement = H->Elements[1];
    int LastElement = H->Elements[H->Size--];
    for (i = 1; i * 2 <= H->Size; i = child)
    {
        //find smaller child
        child = i * 2;
        if (child != H->Size && H->Elements[child] > H->Elements[child + 1])
            child++;

        //Percolate one level
        if (LastElement > H->Elements[child])
            H->Elements[i] = H->Elements[child];
        else
            break;
    }
    H->Elements[i] = LastElement;
    return MinElement;
}




int findKthLargest(int* nums, int numsSize, int k) {
    PriorityQueue H = Initialized(k + 1);
    for (int i = 0; i < numsSize; i++)
    {
        Insert(nums[i], H);
        if (H->Size > k)
            DeleteMin(H);
    }
    return H->Elements[1];
}
```
