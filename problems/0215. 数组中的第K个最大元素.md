# 215. 数组中的第K个最大元素
[力扣题目链接](https://leetcode.cn/problems/kth-largest-element-in-an-array/description/)

---
>Top-K问题
## 思路
用一个优先队列(小顶堆)    
从头到尾遍历数组，数组元素入堆，当堆的长度大于k时，弹出堆顶元素  
最后堆中剩余的即为最大的k个元素，堆顶即为第k大的元素   

### ***C++***
```CPP
class Solution {
public:
    int findKthLargest(vector<int>& nums, int k) {
        priority_queue<int, vector<int>, greater<int>> pri_que;//小顶堆
        //遍历数组，入堆，当堆的长度大于K时，弹出堆顶元素
        for (int i = 0; i < nums.size(); i++)
        {
            pri_que.push(nums[i]);
            if (pri_que.size() > k) 
                pri_que.pop();
        }
        return pri_que.top();
    }
};
```

想办法将前k大的元素放到数组的最左边，第k大的元素为`nums[k - 1]`  
### ***C++***

```CPP
class Solution {
private:
    void swap(int& a, int& b)
    {
        int tmp = a;
        a = b;
        b = tmp;
    }
    void quickSelect(vector<int>& nums, int start, int end, int k)
    {
        if (start >= end) 
            return;
        int left = start, right = end;
        int base = nums[left];
        while (left != right)
        {
            while (left < right && nums[right] <= base)
                right--;
            swap(nums[left], nums[right]);
            while (left < right && nums[left] >= base)
                left++;
            swap(nums[left], nums[right]);
        }
        int size = left - start + 1;
        if (size > k)
            quickSelect(nums, start, left - 1, k);
        else if (size < k)
            quickSelect(nums, left + 1, end, k - size);
        else 
            return;
    }
public:
    int findKthLargest(vector<int>& nums, int k) {
        quickSelect(nums, 0, nums.size() - 1, k);
        return nums[k - 1];
    }
};
```
