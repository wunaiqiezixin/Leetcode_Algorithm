# 239. 滑动窗口最大值
[力扣题目链接](https://leetcode.cn/problems/sliding-window-maximum/description/)

---
>单调队列

## 思路
用C++内置的双端队列deque实现一个单调队列  
需要提供三个API函数  

1. push()
2. pop()
3. getMaxValue()

### ***C++***
```CPP
class MonotonicQueue
{
private:
    deque<int> dq;
public:
    /* add element */
    void Push(int val)
    {
        //val大于队列中的元素的值，出队
        while (!dq.empty() && val > dq.back())
            dq.pop_back();
        
        dq.push_back(val);
    }
    /* pop a element */
    void Pop(int val)
    {
        if (val == dq.front())
            dq.pop_front();
    }
    /* get max value */
    int getMaxValue()
    {
        return dq.front();
    }
};


class Solution {
public:
    vector<int> maxSlidingWindow(vector<int>& nums, int k) {
        MonotonicQueue Window;//单调队列
        vector<int> ans;

        for (int i = 0; i < nums.size(); i++)
        {
            //前k-1个元素进入窗口
            if (i < k - 1)
            {
                Window.Push(nums[i]);
            }
            else
            {
                //窗口移动，加入新元素
                Window.Push(nums[i]);
                //获取最大值
                ans.push_back(Window.getMaxValue());
                //删除旧的元素(窗口左边界)准备移动窗口
                Window.Pop(nums[i - k + 1]);
            }
        }
        return ans;
    }
};
```
