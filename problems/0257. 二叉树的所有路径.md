完整代码
### ***C:***
```C
/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     struct TreeNode *left;
 *     struct TreeNode *right;
 * };
 */
/**
 * Note: The returned array must be malloced, assume caller calls free().
 */
//前序遍历：中-左-右
void getPaths(struct TreeNode* node, int* path, int* pathSize,char** result, int* returnSize)
{   
    /*中*/
    path[(*pathSize)++] = node->val;
    //终止条件
    if (node->left == NULL && node->right == NULL)
    {
        //将路径转化成字符串
        int len = 0;
        char* tmp = (char*)malloc(sizeof(char) * 100);
        for (int i = 0; i < *pathSize - 1; i++)
        {
            len += sprintf(tmp + len, "%d->", path[i]);
        }
        sprintf(tmp + len, "%d", path[*pathSize - 1]);
        result[(*returnSize)++] = tmp;
        return;
    }
    /*左*/
    if (node->left)
    {
        getPaths(node->left, path, pathSize, result, returnSize);
        (*pathSize)--;//回溯
    }
    /*右*/
    if (node->right)
    {
        getPaths(node->right, path, pathSize, result, returnSize);
        (*pathSize)--;//回溯
    }
    return;
}

char** binaryTreePaths(struct TreeNode* root, int* returnSize) {
    char** result = (char**)malloc(sizeof(char*) * 1000);
    if (root == NULL) return result;//默认传入getPaths()函数的节点不为空
    int* path = (int*)malloc(sizeof(int) * 1000);
    *returnSize = 0;
    int pathSize = 0;
    getPaths(root, path, &pathSize, result, returnSize);
    return result;
}
```
### ***C++***
>递归
```CPP
/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
private:
    void traversal(TreeNode* root, vector<string>& ans, vector<int>& path)
    {
        path.push_back(root->val);
        if (root->left == NULL && root->right == NULL)
        {
            string tmp;
            for (int i = 0; i < path.size() - 1; i ++)
            {
                tmp += to_string(path[i]);
                tmp += "->";
            }
            tmp += to_string(path.back());
            ans.push_back(tmp);
            return;
        }
        if (root->left)
        {
            traversal(root->left, ans, path);
            path.pop_back();
        }
        if (root->right)
        {
            traversal(root->right, ans, path);
            path.pop_back();
        }
    }
public:
    vector<string> binaryTreePaths(TreeNode* root) {
        vector<string> ans;
        vector<int> path;
        if (!root) return ans;
        traversal(root, ans, path);
        return ans;
    }
};
```
>隐藏回溯
```CPP
/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
private:
    void traversal(TreeNode* root, vector<string>& ans, string path)
    {
        path += to_string(root->val);//path作用域：该层递归函数(隐藏回溯)
        if (root->left == NULL && root->right == NULL)
        {
            ans.push_back(path);
            return;
        }
        if (root->left) traversal(root->left, ans, path + "->");//隐藏回溯
        if (root->right) traversal(root->right, ans, path + "->");//隐藏回溯
    }
public:
    vector<string> binaryTreePaths(TreeNode* root) {
        vector<string> ans;
        string path;
        traversal(root, ans, path);
        return ans;
    }
};
```

>迭代(维护两个栈)
```CPP
/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    vector<string> binaryTreePaths(TreeNode* root) {
        //维护两个栈
        stack<TreeNode*> treeSt;//节点
        stack<string> pathSt;//路径
        
        vector<string> ans;//路径集
        if (root == NULL) return ans;
        //根节点入栈
        treeSt.push(root);
        pathSt.push(to_string(root->val));
        while (!treeSt.empty())
        {
            //栈顶：中
            TreeNode* node = treeSt.top(); treeSt.pop();
            string path = pathSt.top(); pathSt.pop();
            
            if (!node->left && !node->right) 
            {
                ans.push_back(path);
            }

            if (node->right)//右
            {
                pathSt.push(path + "->" + to_string(node->right->val));
                treeSt.push(node->right);
            }
            if (node->left)//左
            {
                pathSt.push(path + "->" + to_string(node->left->val));
                treeSt.push(node->left);
            }
        }
        return ans;
    }
};
```
