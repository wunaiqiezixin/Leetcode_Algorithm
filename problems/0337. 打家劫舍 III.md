# 337. 打家劫舍 III
[力扣题目链接](https://leetcode.cn/problems/house-robber-iii/description/)

---
>树形dp

## 思路
能打劫到的最大金额为：max(打劫根节点的最大金额, 不打劫根节点的最大金额)  
递归地进行后序遍历  
在每一层递归都有一个长度为2的dp数组，`dp[0]`(不打劫当前节点的最大金额) , `dp[1]` (打劫当前节点的最大金额)  
`dp[0]` = max(不打劫左孩子的最大金额, 打劫左孩子的最大金额) + max(不打劫右孩子的最大金额, 打劫右孩子的最大金额)  
`dp[1]` = 不打劫左孩子的最大金额 + 不打劫右孩子的最大金额 + 当前节点的金额  
然后将dp数组返回给上一层  

### ***C++***
```CPP
/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
private:
    /* 返回一个长度为2的dp数组
        dp[0]: 不打劫当前节点得到的最大金额
        dp[1]: 打劫当前节点得到的最大金额
     */
    vector<int> robTree(TreeNode* node)
    {
        if (node == nullptr) /* 空节点，返回{0, 0} */
            return vector<int>{0, 0};
        /* 
        后序遍历:
            处理单层递归 
        */
        vector<int> leftDp = robTree(node->left);
        vector<int> rightDp = robTree(node->right);
        /* 不打劫当前节点的最大金额val0 */
        int val0 = max(leftDp[0], leftDp[1]) + max(rightDp[0], rightDp[1]);/* 考虑要不要打劫左右节点 */
        /* 打劫当前节点的最大金额val1 */
        int val1 = node->val + leftDp[0] + rightDp[0];/* 左右节点均不能打劫 */
        /* 返回给上一层递归 */
        return vector<int>{val0, val1};
    }
public:
    int rob(TreeNode* root) {
        /* 
        最多能打劫到的金额为：
        打劫root节点的最大金额
        不打劫root节点的最大金额
        两者的最大值
         */
        vector<int> dp = robTree(root);
        return max(dp[0], dp[1]);
    }
};
```
