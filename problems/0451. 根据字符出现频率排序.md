# 451. 根据字符出现频率排序
[力扣题目链接](https://leetcode.cn/problems/sort-characters-by-frequency/description/)

---
## 思路
用`map`统计字符的频率   
用`set`按字符频率从高到低对字符排序   

### ***C++***
```CPP
class Solution {
private:
    struct mycomprison {
        bool operator() (const pair<char, int> & a1, const pair<char, int> & a2) const {
            if (a1.second == a2.second)//频率相同，字典序小的在前
                return a1.first < a2.first;
            return a1.second > a2.second;//频率高的在前
        }
    };
public:
    string frequencySort(string s) {
        unordered_map<char, int> map;//map统计频率
        for (int i = 0; i < s.size(); i++)
        {
            map[s[i]]++;
        }
        set<pair<char, int>, mycomprison> st;//根据词频进行排序
        for (unordered_map<char, int>::iterator i = map.begin(); i != map.end(); i++)
        {
            st.insert(*i);
        }
        string ans;
        //收集结果
        for (set<pair<char, int>, mycomprison>::iterator i = st.begin(); i != st.end(); i++)
        {
            for (int k = 0; k < i->second; k++)
            {
                ans.push_back(i->first);
            }
        }
        return ans;
    }
};
```
