# 474. Ones and Zeroes
[力扣题目链接](https://leetcode.cn/problems/ones-and-zeroes/description/)

---

### C++
```CPP
class Solution {
public:
    int findMaxForm(vector<string>& strs, int m, int n) {
        /* dp[i][j]:
        有i个0，j个1的strs的最大子集长度
        */
        vector<vector<int>> dp(m + 1, vector<int>(n + 1, 0));

        //dp[i][j] = max(dp[i - x][j - y] + 1, dp[i][j]
        int x, y;
        for (string str : strs)
        {
            x = y = 0;
            for (char ch : str)
            {
                if (ch == '1')
                    y++;
                else
                    x++;
            }
            for (int i = m; i >= x; i--)
            {
                for (int j = n; j >= y; j--)
                {
                    dp[i][j] = max(dp[i][j], dp[i - x][j - y] + 1);
                }
            }
        }
        return dp[m][n];
    }
};
```
