# 494. Target Sum
[力扣题目链接](https://leetcode.cn/problems/target-sum/description/)

---
>回溯法
```CPP
class Solution {
private:
    int result;
    void backtracking(vector<int>& nums, int val, int startIndex)
    {
        if (val < 0)
            return;
        if (val == 0)
            result++; /* 不能return, 因为i后面可能为0 */
        for (int i = startIndex; i < nums.size(); i++)
            backtracking(nums, val - nums[i], i + 1);
    }
public:
    int findTargetSumWays(vector<int>& nums, int target) {
        int Sum = accumulate(nums.begin(), nums.end(), 0);
        if (Sum < target || (Sum + target) % 2 == 1)
            return 0;
        
        int val = (Sum + target) / 2;
        result = 0;
        backtracking(nums, val, 0);
        return result;
    }
};
```
