/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     struct TreeNode *left;
 *     struct TreeNode *right;
 * };

int getMax(int* nums, int size)
{
    //çº¿æ€§æŸ¥æ‰¾
    int mx = nums[0];
    int index = 0;
    for (int i = 1; i < size; i++)
    {
        if (mx < nums[i])
        {
            mx = nums[i];
            index = i;
        }
    }
    return index;
}

    //ç»ˆæ­¢æ¡ä»¶
    if (numsSize == 0) return NULL;
    //æ„å»ºä¸€ä¸ªèŠ‚ç‚¹
    int index = getMax(nums, numsSize);

    root->val = val;
    
    //"åˆ†å‰²æ•°ç»„"

    return root;
}
