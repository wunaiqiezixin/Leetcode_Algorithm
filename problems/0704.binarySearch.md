##左闭右闭：
### **C:**

```c
int search(int* nums, int numsSize, int target) {
    int left = 0,right = numsSize - 1;
    int mid;
    while (left <= right)
    {
        mid = left + (right - left);
        if (nums[mid] == target)
        {
            return mid;
        }
        else if (nums[mid] < target)
        {
            left = mid + 1;
        }
        else
        {
            right = mid - 1;
        }
    }
    return -1;
}
```
