# 707. 设计链表
[力扣题目链接](https://leetcode.cn/problems/design-linked-list/description/)
* 考察基本功
* 下面给出两种C语言的实现：
* 单向链表(不维护头尾结点，设定虚拟结点)
* 双向链表(不维护头尾结点、设定虚拟结点)
## 单向链表
### ***C***
```C
typedef struct _Node{
    int val;
    struct _Node* next;
}Node;


typedef struct {
    int size;
    Node* dummyhead;
} MyLinkedList;


MyLinkedList* myLinkedListCreate() {
    MyLinkedList* obj = (MyLinkedList*)malloc(sizeof(MyLinkedList));
    obj->size = 0;
    Node* head = (Node*)malloc(sizeof(Node));
    head->next = (void*)0;
    obj->dummyhead = head;
    return obj;
}

int myLinkedListGet(MyLinkedList* obj, int index) {
    if (index < 0 || index >= obj->size) return -1;

    Node* ptr = obj->dummyhead;
    while (index-- >= 0)
    {
        ptr = ptr->next;
    }
    return ptr->val;
}

void myLinkedListAddAtHead(MyLinkedList* obj, int val) {
    Node* ptr = obj->dummyhead;
    Node* new = (Node*)malloc(sizeof(Node));
    new->next = ptr->next;
    new->val = val;
    ptr->next = new;
    obj->size++;
}

void myLinkedListAddAtTail(MyLinkedList* obj, int val) {
    Node* ptr = obj->dummyhead;
    while (ptr->next)
    {
        ptr = ptr->next;
    }
    Node* new = (Node*)malloc(sizeof(Node));
    new->val = val;
    new->next = NULL;
    ptr->next = new;
    obj->size++;
}

void myLinkedListAddAtIndex(MyLinkedList* obj, int index, int val) {
    if (index < 0 || index > obj->size) return; //index可以等于size
    Node* ptr = obj->dummyhead;
    while (index--)
    {
        ptr = ptr->next;
    }
    Node* new = (Node*)malloc(sizeof(Node));
    new->val = val;
    new->next = ptr->next;
    ptr->next = new;
    obj->size++;
}

void myLinkedListDeleteAtIndex(MyLinkedList* obj, int index) {
    if (index < 0 || index >= obj->size) return;
    Node* ptr = obj->dummyhead;
    while (index--)
    {
        ptr = ptr->next;
    }
    Node* cur = ptr->next;
    ptr->next = ptr->next->next;
    free(cur);
    obj->size--;
}

void myLinkedListFree(MyLinkedList* obj) {
    Node* ptr = obj->dummyhead;
    while (ptr)
    {
        Node* cur = ptr;
        ptr = ptr->next;
        free(cur);
    }
    free(obj);
}

/**
 * Your MyLinkedList struct will be instantiated and called as such:
 * MyLinkedList* obj = myLinkedListCreate();
 * int param_1 = myLinkedListGet(obj, index);
 
 * myLinkedListAddAtHead(obj, val);
 
 * myLinkedListAddAtTail(obj, val);
 
 * myLinkedListAddAtIndex(obj, index, val);
 
 * myLinkedListDeleteAtIndex(obj, index);
 
 * myLinkedListFree(obj);
*/
```
## 双向链表
## ***C***
```C
typedef struct _Node{
    struct _Node* prev;
    struct _Node* next;
    int val;
}Node;


typedef struct {
    Node* dummyhead;
    int size;
} MyLinkedList;


MyLinkedList* myLinkedListCreate() {
    MyLinkedList* obj = (MyLinkedList*)malloc(sizeof(MyLinkedList));
    Node* head = (Node*)malloc(sizeof(Node));
    head->prev = NULL;
    head->next = NULL;
    obj->dummyhead = head;
    obj->size = 0;
    return obj;
}

int myLinkedListGet(MyLinkedList* obj, int index) {
    if (index < 0 || index >= obj->size) return -1;

    Node* ptr = obj->dummyhead;
    while (index-- >= 0)
    {
        ptr = ptr->next;
    }
    return ptr->val;
}

void myLinkedListAddAtHead(MyLinkedList* obj, int val) {
    Node* ptr = obj->dummyhead;
    Node* new = (Node*)malloc(sizeof(Node));
    new->val = val;
    new->prev = ptr;
    new->next = ptr->next;
    if (ptr->next != NULL)
    {
        ptr->next->prev = new;
    }
    ptr->next = new;
    obj->size++;
}

void myLinkedListAddAtTail(MyLinkedList* obj, int val) {
    Node* ptr = obj->dummyhead;
    while (ptr->next)
    {
        ptr = ptr->next;
    }
    Node* new = (Node*)malloc(sizeof(Node));
    new->val = val;
    new->prev = ptr;
    new->next = NULL;
    ptr->next = new;
    obj->size++;
}

void myLinkedListAddAtIndex(MyLinkedList* obj, int index, int val) {
    if (index < 0 || index > obj->size) return; //注意index可以等于size

    Node* ptr = obj->dummyhead;
    while (index--)
    {
        ptr = ptr->next;
    }
    Node* new = (Node*)malloc(sizeof(Node));
    new->val = val;
    new->prev = ptr;
    new->next = ptr->next;
    if (ptr->next)
    {
        ptr->next->prev = new;
    }
    ptr->next = new;
    obj->size++;
}

void myLinkedListDeleteAtIndex(MyLinkedList* obj, int index) {
    if (index < 0 || index >= obj->size) return;

    Node* ptr = obj->dummyhead;
    while (index--)
    {
        ptr = ptr->next;
    }
    Node* cur = ptr->next;
    if (ptr->next->next)
    {
        ptr->next->next->prev = ptr;
    }
    ptr->next = ptr->next->next;
    free(cur);
    obj->size--;
}

void myLinkedListFree(MyLinkedList* obj) {
    Node* ptr = obj->dummyhead;
    while (ptr)
    {
        Node* cur = ptr;
        ptr = ptr->next;
        free(cur);
    }
    free(obj);
}

/**
 * Your MyLinkedList struct will be instantiated and called as such:
 * MyLinkedList* obj = myLinkedListCreate();
 * int param_1 = myLinkedListGet(obj, index);
 
 * myLinkedListAddAtHead(obj, val);
 
 * myLinkedListAddAtTail(obj, val);
 
 * myLinkedListAddAtIndex(obj, index, val);
 
 * myLinkedListDeleteAtIndex(obj, index);
 
 * myLinkedListFree(obj);
*/
```
