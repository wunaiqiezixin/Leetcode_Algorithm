# 968. 监控二叉树
[力扣题目链接](https://leetcode.cn/problems/binary-tree-cameras/description/)

---
## 思路
一个摄像头可以覆盖住上中下三层，所以叶子节点不要放摄像头  
于是自底向上(后序遍历)  

节点的状态有三种：
1. 节点放上了摄像头
2. 节点被摄像头覆盖了(有覆盖)
3. 节点没有被节点覆盖(无覆盖)

### ***C++***
```CPP
/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
private:
    int result = 0;
    /* ********
    返回0：节点有摄像头
    返回1：节点有覆盖
    返回2：节点无覆盖
    * ********/
    int traversal(TreeNode* node)
    {
        if (node == NULL)
            return 1;//空节点返回有覆盖
        int left = traversal(node->left);
        int right = traversal(node->right);
        //情况1：左右孩子都有覆盖(node可能为根节点)
        if (left == 1 && right == 1)
            return 2;//节点无覆盖
        //情况2：左右孩子至少有一个无覆盖
        if (left == 2 || right == 2)
        {
            result++;
            return 0;//放上摄像头
        }
        //情况3：左右孩子至少有一个有摄像头
        if (left == 0 || right == 0)
            return 1;//节点有覆盖
        return -1;
    }
public:
    int minCameraCover(TreeNode* root) {
        int ret = traversal(root);
        //如果根节点无覆盖
        if (ret == 2)
            result++;
        return result;
    }
};
```
